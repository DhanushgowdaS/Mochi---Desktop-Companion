#include <SPI.h>
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h> // Switched for 0.96" OLED 
#include <FluxGarage_RoboEyes.h>

#define i2c_Address 0x3c //initialize with the I2C addr 0x3C Typically eBay OLED's
//#define i2c_Address 0x3d //initialize with the I2C addr 0x3D Typically Adafruit OLED's
#define SCREEN_WIDTH 128 // OLED display width, in pixels
#define SCREEN_HEIGHT 64 // OLED display height, in pixels
#define OLED_RESET -1   //   QT-PY / XIAO

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);
RoboEyes<Adafruit_SSD1306> roboEyes(display); // create RoboEyes instance

// --- Touch Sensor Logic Setup ---
const int touchPin = 1;     // Touch sensor connected to Pin 1
int moodCounter = 0;        // To track which mood we are on
bool lastTouchState = LOW;  // To detect the "tap" (edge detection)

void setup()   {
  Serial.begin(9600);
  pinMode(touchPin, INPUT); // Initialize the touch sensor pin

  // Custom I2C pins for ESP32-C3
  Wire.begin(21, 20); 

  delay(250); // wait for the OLED to power up
  display.begin(SSD1306_SWITCHCAPVCC, i2c_Address); // Address 0x3C default
  //display.setContrast (0); // dim display

  // Startup robo eyes
  roboEyes.begin(SCREEN_WIDTH, SCREEN_HEIGHT, 100); // screen-width, screen-height, max framerate

  // Define some automated eyes behaviour
  roboEyes.setAutoblinker(ON, 3, 2); 
  roboEyes.setIdleMode(ON, 2, 2); 
  
  
} // end of setup


void loop() {
  // Check touch sensor state
  bool currentTouchState = digitalRead(touchPin);

  // If touch is detected and it's a NEW touch (not holding it down)
  if (currentTouchState == HIGH && lastTouchState == LOW) {
    moodCounter++; // Move to next mood
    if (moodCounter > 3) moodCounter = 0; // Reset back to first mood

    // Reset position to center whenever mood changes
    roboEyes.setPosition(DEFAULT);

    // Apply the mood based on the counter
    if (moodCounter == 0) roboEyes.setMood(DEFAULT);
    else if (moodCounter == 1) roboEyes.setMood(TIRED);
    else if (moodCounter == 2) roboEyes.setMood(ANGRY);
    else if (moodCounter == 3) roboEyes.setMood(HAPPY);
    
    delay(50); // Small debounce delay
  }
  lastTouchState = currentTouchState;

  roboEyes.update(); // update eyes drawings
  // Dont' use delay() here in order to ensure fluid eyes animations.
 }
